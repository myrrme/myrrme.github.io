#                   Переменные
# Переменная это хранилище для какого либо типа данных
# Варианты типа данных: int - целочисленный тип
#                       float - вещественный тип
#                       str - строковый тип данных
#
# Из чего состоит переменная:
# Имя переменной, данные которые хранит переменная.
#
# Правила при создании имени переменной:
# Имена переменных могут быть любой длины,
# могут иметь заглавные и
# строчные буквы(от A до Z, от a до z),
# цифру(0-9) и символ подчеркивания(_).
#
# Переменные предназначены для хранения данных.
# Название переменной в Python должно начинаться с
# алфавитного символа или со знака подчеркивания
# и может содержать алфавитно-цифровые символы
# и знак подчеркивания.
# И кроме того, название переменной не должно совпадать
# с названием ключевых слов языка Python.
#
# Ключевых слов не так много, их легко запомнить:
# False      await      else       import     pass
# None       break      except     in         raise
# True       class      finally    is         return
# and        continue   for        lambda     try
# as         def        from       nonlocal   while
# assert     del        global     not        with
# async      elif       if         or         yield
#
a = 5        # переменная имеет целочисленный тип данных
b = 5.5      # переменная имеет вещественный тип данных
c = "Привет" # переменная имеет строковый тип данных

print(a)     # вывод переменной " a "
print(b)     # вывод переменной " b "
print(c)     # вывод переменной " c "

input() #команда для ввода информации с клавиатуры

# команда input() работает вместе с переменной
# Варианты написания:
# a = input() -> по умолчанию все что мы введём с клавиатуры
#                будет иметь строковый тип данных

a = input("Введите данные: ") # -> комментарий при вводе
a = input('Введите данные: ') # -> комментарий при вводе

# a = int(input()) - > введеные с клавиатуры данные
#                      примут целочисленное значение
# Примечание: если ввести символы или текст возникнет ошибка
#
# a = float(input()) - > введеные с клавиатуры данные
#                      примут вещественное значение
# Примечание: если ввести символы или текст возникнет ошибка
#
# a = str(input()) - > введеные с клавиатуры данные
#                      примут строковое значение значение
# Примечание: можно вводить что угодно
#
# Элементарные операции над переменными
# 1) Операция сложения. Знак операции " + "
# 2) Операция вычитание. Знак операции " - "
# 3) Операция умножения. Знак операции " * "
# 4) Операция деления. Знак операции " / "
#   Примечание: в Python существует 3 типа деления
#               более подробно в уроке " Типы деления "
# 5) Операция возведение в степень. Знак операции " ** "

# Примечание: Элементарные математические операции можно
#             проводить только над типами данных int и float

# Примеры элементарных операций
a = 7
b = 17
print(a + b)   # умножение
print(a - b)   # вычитание
print(a * b)   # умножение
print(a / b)   # деление
print(a ** b)  # возведение в степень

"""Даны натуральное число n и целые числа a1, a2, …, an,
среди которых имеются числа, равные 10. Найти:
а) номер последнего из них;
б) номер первого из них."""
n = int(input('Введите количество элементов a для ввода: '))
for i in range(1,n+1):
  print("Введите элемент A",i,": ")
  a = int(input())
  if a == 10:
    countA = i
    break #break для буквы б)
print(countA)
